	<!--include some associate libraries.-->
	<script src="https://webduino.io/components/webcomponentsjs/webcomponents.js"></script>
	<link rel="import" href="https://webduino.io/components/webduino/web-arduino.html" />
	<link rel="import" href="https://webduino.io/components/webduino/wa-led.html" />
	<!--add Webduino tag and led tag.-->
	<div id="light" class="off">
		<img src="http://i.imgur.com/T5H4MHE.png" />
		<img src="http://i.imgur.com/8qFj2Ou.png" />
	</div>
	<web-arduino id="board" device="你的 device 名稱">
		<wa-led id="led" pin="10"></wa-led>
	</web-arduino>



	/*set the image css*/
	#light img {
		width:100%;
		display:none;
	}
	#light.off img:first-child {
		display:inline-block;
	}
	#light.on img:last-child {
		display:inline-block;
	}



	//Using native JS light the led
	window.addEventListener('WebComponentsReady', function () {
		var board = document.getElementById('board'),
		light = document.getElementById('light');

		board.on('ready',function ready() {
			var led = document.getElementById('led');

			light.addEventListener('click', function() {
				if(light.className == 'on') {
					led.off();     // 讓 LED 熄滅
					light.className = 'off';
				}
				else {
					led.on();     // 讓 LED 亮起
					light.className = 'on';
				}
			}, false);
		});

	}, false);

	//I want to use jQuery.
	window.addEventListener('WebComponentsReady', function () {
		var board = document.getElementById('board'),

		board.on('ready',function ready() {
			var led = document.getElementById('led');

	<!--just inclue one additional plugin and jQery core libraries-->
	<--include some associate libraries.-->
	<script src="https://webduino.io/components/webcomponentsjs/webcomponents.js"></script>
	<script src="//code.jquery.com/jquery-1.11.3.min.js"></script>
	<script src="js/jqduino.js"></script>
	<link rel="import" href="https://webduino.io/components/webduino/web-arduino.html" />
	<link rel="import" href="https://webduino.io/components/webduino/wa-led.html" />


	
		//And you can use following code !
		$(function() {
			$("#board").attr("device", "3KLK");
				$(document).web_components_ready(function () {
					$("#board").board_ready(function() {
						$("#light").click(function(event) {
						event.preventDefault();
						if($("#light").attr("class") == "on") {
							$("#led").set_led("off");
							$("#light").attr("class", "off");
						}
						else {
							$("#led").set_led("on");
							$("#light").attr("class", "on");
						}
					});
				});
			});
	
		});
